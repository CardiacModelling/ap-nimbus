import { Injectable } from '@angular/core';
import { HttpClient, HttpParams } from '@angular/common/http';
import { HttpHeaders } from '@angular/common/http';
import { HttpErrorResponse, HttpResponse } from '@angular/common/http';

import { Observable, of, from, throwError } from 'rxjs';
import { catchError, retry, map } from 'rxjs/operators';

import { ApPredictInput } from './ap-predict-input';
import { ApPredictOutput } from './ap-predict-output';
import { EnvService } from './env.service';

/**
 * JSON content type HTTP header.
 */
const httpHeaderJSON = {
  headers: new HttpHeaders({ 
    'Content-Type': 'application/json'
  })
}

/**
 * text/plain content type HTTP header.
 */
const headers = new HttpHeaders().set('Content-Type', 'text/plain; charset=utf-8');

export enum GetOp {
  PROGRESS,
  STOP_INDICATOR,
  VOLTAGE_TRACES
}

/**
 * Service for REST invocations of the `app-manager` application.
 */
@Injectable({
  providedIn: 'root'
})
export class AppredictRestService {

  // Set the default app-manager endpoint.
  apPredictRestUrl: string = 'http://localhost:8080/';

  /**
   * Initialising constructor.
   * 
   * @param httpClient HTTP client.
   * @param envService Environment variable service.
   */
  constructor(private httpClient: HttpClient, private envService: EnvService) {
    let envUrl: string = envService.apiUrl;
    if (envUrl != '') {
      this.apPredictRestUrl = envUrl;
    }
  }

  /**
   * Handle backend errors.
   * 
   * @param error Error to handle.
   * @link https://angular.io/guide/http#getting-error-details 
   */
  private handleError(error: HttpErrorResponse) {
    if (error.error instanceof ErrorEvent) {
      // A client-side or network error occurred. Handle it accordingly.
      console.error('An error occurred:', error.error.message);
    } else {
      // The backend returned an unsuccessful response code.
      // The response body may contain clues as to what went wrong,
      console.error(
        `Backend returned code ${error.status}, ` +
        `body was: ${error.error}`);
    }
    // return an observable with a user-facing error message
    return throwError(
      'Something bad happened; please try again later.');
  };

  /**
   * `POST` data to start an `ApPredict` simulation on the `app-manager` application.
   * 
   * @param apPredictInput `ApPredict` input data.
   * @return `ApPredict` invocation/simulation output.
   */
  invokeApPredict(apPredictInput: ApPredictInput): Observable<ApPredictOutput> {
    return this.httpClient.post<ApPredictOutput>(this.apPredictRestUrl, apPredictInput,
                                                 httpHeaderJSON)
                          .pipe(catchError(this.handleError))
  }

  /**
   * Query the `app-manager` application to retrieve simulation data.
   * 
   * @param simulationId Simulation identifier to query.
   * @param getOpt Nature of query, e.g. retrieve voltage trace, progress, etc. data.
   * @return `ApPredict` invocation/simulation output.
   */
  queryApPredict(simulationId: string,
                 getOpt: GetOp): Observable<ApPredictOutput> {
    let queryURL: string = this.apPredictRestUrl + "?id=" + simulationId;

    switch (getOpt) {
      case GetOp.VOLTAGE_TRACES:
        queryURL += "&op=v";
        return this.httpClient.get(queryURL, { headers, responseType: 'text' })
                              .pipe(map(response => {
                                          return new ApPredictOutput(simulationId, false,
                                                                     '', response);
                                        },catchError(this.handleError) ));
      case GetOp.STOP_INDICATOR:
        queryURL += "&op=s";
        return this.httpClient.get(queryURL, { headers, responseType: 'text' })
                              .pipe(map(response => {
                                          var stopped = (response == "STOPPED") ? true : false;
                                          return new ApPredictOutput(simulationId, stopped,
                                                                     '', '');
                                        },catchError(this.handleError) ));
      default:
        let errorMessage: string = '~queryApPredict() : Unrecognised operation ' + getOpt;
        console.error(errorMessage);
        return of(new ApPredictOutput(simulationId, false, errorMessage, ''));
    }
  }
}
